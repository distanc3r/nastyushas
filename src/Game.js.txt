import React, { useState } from "react";
import { Card, CardContent } from "@/components/ui/card";
import { Button } from "@/components/ui/button";
import { motion } from "framer-motion";
import steps from "./steps"; // steps.js dosyasÄ±nÄ± import et

const Game = () => {
  const [currentStep, setCurrentStep] = useState(0);
  const [message, setMessage] = useState("");
  const [showFinalVideo, setShowFinalVideo] = useState(false);

  const step = steps[currentStep];

  const handleAnswer = (selectedIndex) => {
    if (selectedIndex === step.answer) {
      setMessage(step.message);
      if (currentStep + 1 < steps.length) {
        setTimeout(() => {
          setCurrentStep(currentStep + 1);
          setMessage(""); // MesajÄ± sÄ±fÄ±rla
        }, 2500); // 2.5 saniye sonra bir sonraki soruya geÃ§
      } else {
        setShowFinalVideo(true); // Video ekranÄ±nÄ± gÃ¶ster
      }
    } else {
      setMessage("Try again, kotik ğŸ’›");
    }
  };

  if (showFinalVideo) {
    return (
      <div className="final-video">
        <h2>ğŸ¥³ Congratulations! You have finished the game!</h2>
        <a href={step.finalVideo} target="_blank" rel="noopener noreferrer">
          <button>Click here to watch your special video ğŸ¥</button>
        </a>
      </div>
    );
  }

  return (
    <motion.div
      className="game-container"
      initial={{ opacity: 0 }}
      animate={{ opacity: 1 }}
      transition={{ duration: 0.5 }}
    >
      <Card>
        <CardContent>
          <h3>Step {currentStep + 1}</h3>
          <p>{step.question}</p>
          {step.image && <img src={step.image} alt="Step Image" />}
          <div className="options">
            {step.options.map((option, index) => (
              <Button key={index} onClick={() => handleAnswer(index)}>
                {option}
              </Button>
            ))}
          </div>
          {message && <p>{message}</p>}
        </CardContent>
      </Card>
    </motion.div>
  );
};

export default Game;
